async function createProposal(){

  btn.addEventListener('click', async ()=> {
    let arg = {
        propertyId: 0,
        what: {
            Governance: {

                    Create:
                        [{

                            category = document.querySelector('input[name="category"]:checked')?.value,

                            startAt = Math.floor(new Date(document.getElementById("start-time").value).getTime() / 1000),
                            actions: []
                        }]
                }
            }
        }

    } 

  })
}
